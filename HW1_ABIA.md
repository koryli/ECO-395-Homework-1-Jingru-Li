``` r
#For DAL
Q5 = ABIA%>%
  filter(Dest == "DAL")%>%
  na.omit()%>%
  group_by(Month,DayofMonth) %>%
  summarise(delay1 = mean(ArrDelay),
            delay2 = mean(DepDelay)) %>%
  mutate(delaymean = delay1+delay2)
```
